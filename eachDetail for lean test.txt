  checkRole() async {
    QuerySnapshot<Map<String, dynamic>> roles = await FirebaseFirestore.instance
        .collection("old")
        .where("user_id", isEqualTo: FirebaseAuth.instance.currentUser!.uid)
        .get();
    if (roles.docs[0]['class_role'] == "learn" &&
        roles.docs[0]['course_name'] == "${widget.nameCourse.name}") {
      setState(() {
        this.student = true;
      });
    }
  }
